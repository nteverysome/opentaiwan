📄 opentaiwan.tsx 
import React, { useState, useEffect } from 'react';

export default function OpenTaiwan() {
  const [apiData, setApiData] = useState({
    health: null,
    status: null,
    quota: null
  });

  useEffect(() => {
    fetchApiData();
  }, []);

  const fetchApiData = async () => {
    try {
      const [healthRes, statusRes, quotaRes] = await Promise.all([
        fetch('http://localhost:3002/api/health'),
        fetch('http://localhost:3002/api/status'),
        fetch('http://localhost:3002/api/quota')
      ]);

      setApiData({
        health: await healthRes.json(),
        status: await statusRes.json(),
        quota: await quotaRes.json()
      });
    } catch (error) {
      console.error('API 調用失敗:', error);
    }
  };

  return (
    <div className="min-h-screen bg-gradient-to-br
                   from-blue-900 via-purple-900 to-indigo-900">
      <div className="container mx-auto px-6 py-12">
        <h1 className="text-4xl font-bold text-white mb-8
                      text-center animate-fade-in">
          OpenTaiwan
        </h1>

        <div className="grid grid-cols-1 md:grid-cols-3 gap-6">
          {/* 配額狀態卡片 */}
          <div className="bg-white/10 backdrop-blur-md rounded-xl
                         p-6 border border-white/20 hover:bg-white/20
                         transition-all duration-300">
            <h3 className="text-xl font-semibold text-white mb-4">
              配額狀態
            </h3>
            <div className="space-y-2">
              <div className="flex justify-between text-gray-200">
                <span>已使用</span>
                <span>{apiData.quota?.quota?.used || 0}/
                      {apiData.quota?.quota?.total || 10}</span>
              </div>
              <div className="w-full bg-gray-600 rounded-full h-2">
                <div className="bg-blue-400 h-2 rounded-full"
                     style={{width: `${((apiData.quota?.quota?.used || 0) /
                            (apiData.quota?.quota?.total || 10)) * 100}%`}}>
                </div>
              </div>
            </div>
          </div>

          {/* 系統狀態卡片 */}
          <div className="bg-white/10 backdrop-blur-md rounded-xl
                         p-6 border border-white/20 hover:bg-white/20
                         transition-all duration-300">
            <h3 className="text-xl font-semibold text-white mb-4">
              系統狀態
            </h3>
            <div className="space-y-2">
              <div className="flex items-center gap-2">
                <div className="w-3 h-3 bg-green-400 rounded-full"></div>
                <span className="text-gray-200">
                  {apiData.status?.status || 'Loading...'}
                </span>
              </div>
              <div className="text-sm text-gray-300">
                運行時間: {Math.floor(apiData.status?.uptime || 0)}s
              </div>
            </div>
          </div>

          {/* API 健康狀態卡片 */}
          <div className="bg-white/10 backdrop-blur-md rounded-xl
                         p-6 border border-white/20 hover:bg-white/20
                         transition-all duration-300">
            <h3 className="text-xl font-semibold text-white mb-4">
              API 健康
            </h3>
            <div className="space-y-2">
              <div className="flex items-center gap-2">
                <div className="w-3 h-3 bg-green-400 rounded-full"></div>
                <span className="text-gray-200">
                  {apiData.health?.status || 'Checking...'}
                </span>
              </div>
              <button
                onClick={fetchApiData}
                className="text-blue-400 hover:text-blue-300 text-sm">
                🔄 刷新狀態
              </button>
            </div>
          </div>
        </div>
      </div>
    </div>
  );
}
