interface User {
  id: string;
  name: string;
  email: string;
  avatar?: string;
}

export function UserProfile({ user }: { user: User }) {
  return (
    <div className="user-profile">
      <img
        src={user.avatar || '/default-avatar.png'}
        alt={user.name}
        className="avatar"
      />
      <h2>{user.name}</h2>
      <p>{user.email}</p>
    </div>
  );
}
